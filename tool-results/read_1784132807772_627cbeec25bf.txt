    74тЖТ    74тЖТ  const mins = Math.floor(diff / 60000);
    75тЖТ    75тЖТ  if (mins < 60) return `${mins}m ago`;
    76тЖТ    76тЖТ  const hrs = Math.floor(mins / 60);
    77тЖТ    77тЖТ  if (hrs < 24) return `${hrs}h ago`;
    78тЖТ    78тЖТ  const days = Math.floor(hrs / 24);
    79тЖТ    79тЖТ  return `${days}d ago`;
