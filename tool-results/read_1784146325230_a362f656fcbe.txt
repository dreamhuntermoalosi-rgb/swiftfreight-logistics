   173→function formatCurrency(value: number) {

   177→function timeAgo(dateStr: string): string {
   178→  const diff = Date.now() - new Date(dateStr).getTime();
   179→  const hours = Math.floor(diff / 3600000);
   180→  if (hours < 1) return `${Math.floor(diff / 60000)}m ago`;
   181→  if (hours < 24) return `${hours}h ago`;
   182→  const days = Math.floor(hours / 24);
   183→  return `${days}d ago`;
   184→}

   205→// ============ KPI Card ============
   206→function KpiCard({
   207→  title,
   208→  value,
   209→  growth,
   210→  icon: Icon,
   211→  prefix = '',
   212→  suffix = '',
   213→  isNegative = false,
   214→  iconBg = 'bg-emerald-50 dark:bg-emerald-950/50',
   215→  iconColor = 'text-emerald-600',
   216→  pulse = false,
   217→}: {
   218→  title: string;
   219→  value: string;
   220→  growth?: number;
   221→  icon: React.ComponentType<{ className?: string }>;
   222→  prefix?: string;
   223→  suffix?: string;
   224→  isNegative?: boolean;
   225→  iconBg?: string;
   226→  iconColor?: string;
   227→  pulse?: boolean;
   228→}) {
   229→  const isPositive = growth !== undefined && growth >= 0;
   230→  return (
   231→    <Card className="relative overflow-hidden transition-all duration-300 hover:shadow-lg hover:-translate-y-[2px] hover:border-emerald-300 dark:hover:border-emerald-700">
   232→      {/* Gradient top border */}
   233→      <div className="absolute top-0 left-0 right-0 h-1 bg-gradient-to-r from-primary to-teal-500" />

   236→          <div className="space-y-1">

   238→              {title}
   239→              {pulse && <span className="h-2 w-2 rounded-full bg-emerald-500 animate-pulse" />}
   240→            </p>
   241→            <p className="text-2xl font-bold tracking-tight">
   242→              {prefix}{value}{suffix}
   243→            </p>
   244→          </div>
   245→          <div className="flex h-12 w-12 shrink-0 items-center justify-center rounded-full bg-gradient-to-br from-primary/10 to-teal-500/10">
   246→            <Icon className={`h-6 w-6 ${iconColor}`} />
   247→          </div>
   248→        </div>
   249→        {growth !== undefined && (
   250→          <div className="mt-3 flex items-center gap-1.5 rounded-full bg-primary/5 px-2.5 py-1 w-fit">
   251→            {isPositive ? (
   252→              <ArrowUpRight className="h-3.5 w-3.5 text-emerald-600" />
   253→            ) : (
   254→              <TrendingDown className={`h-3.5 w-3.5 ${isNegative ? 'text-red-500' : 'text-emerald-600'}`} />
   255→            )}
   256→            <span
   257→              className={`text-sm font-semibold ${
   258→                isNegative ? (isPositive ? 'text-red-500' : 'text-emerald-600') : 'text-emerald-600'
   259→              }`}
   260→            >
   261→              {isPositive ? '+' : ''}{growth}%
   262→            </span>
   263→            <span className="text-xs text-muted-foreground">vs last period</span>
   264→          </div>
   265→        )}
   266→      </CardContent>
   267→    </Card>
   268→  );
   269→}

   309→  // Find customer matching the logged-in user

   311→    return customers.find((c) => c.name === currentUser?.name) || customers[0];
   312→  }, [currentUser]);
   313→
   314→  const customerDeliveries = useMemo(
   315→    () => deliveries.filter((d) => d.customerId === customerRecord.id),
   316→    [customerRecord.id]
   317→  );
   318→
   319→  const activeShipments = useMemo(
   320→    () => customerDeliveries.filter((d) => d.status !== 'delivered' && d.status !== 'cancelled' && d.status !== 'returned'),
   321→    [customerDeliveries]
   322→  );
   323→
   324→  const deliveredShipments = useMemo(
   325→    () => customerDeliveries.filter((d) => d.status === 'delivered'),
   326→    [customerDeliveries]
   327→  );
   328→
   329→  const totalSpent = useMemo(
   330→    () => customerDeliveries.reduce((sum, d) => sum + (d.paidAmount || 0), 0),
   331→    [customerDeliveries]
   332→  );
   333→
   334→  const recentShipments = useMemo(
   335→    () => [...customerDeliveries].sort((a, b) => new Date(b.createdAt).getTime() - new Date(a.createdAt).getTime()).slice(0, 5),
