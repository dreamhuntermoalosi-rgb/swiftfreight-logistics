    22тЖТ    22тЖТfunction exportCSV(data: Record<string, unknown>[], filename: string) {
    23тЖТ    23тЖТ  if (data.length === 0) return;
    24тЖТ    24тЖТ  const headers = Object.keys(data[0]);
    25тЖТ    25тЖТ  const csv = [
    26тЖТ    26тЖТ    headers.join(','),
    27тЖТ    27тЖТ    ...data.map(row => headers.map(h => `"${String(row[h] ?? '').replace(/"/g, '""')}"`).join(','))
    28тЖТ    28тЖТ  ].join('\n');
    29тЖТ    29тЖТ  const blob = new Blob([csv], { type: 'text/csv;charset=utf-8;' });
    30тЖТ    30тЖТ  const url = URL.createObjectURL(blob);
    31тЖТ    31тЖТ  const a = document.createElement('a');
    32тЖТ    32тЖТ  a.href = url;
    33тЖТ    33тЖТ  a.download = filename;
    34тЖТ    34тЖТ  a.click();
    35тЖТ    35тЖТ  URL.revokeObjectURL(url);
    36тЖТ    36тЖТ  toast.success(`Exported ${data.length} records to ${filename}`);
    37тЖТ    37тЖТ}
