   183тЖТ   183тЖТ  const diff = Date.now() - new Date(dateStr).getTime();
   184тЖТ   184тЖТ  const hours = Math.floor(diff / 3600000);
   185тЖТ   185тЖТ  if (hours < 1) return `${Math.floor(diff / 60000)}m ago`;
   186тЖТ   186тЖТ  if (hours < 24) return `${hours}h ago`;
   187тЖТ   187тЖТ  const days = Math.floor(hours / 24);
   188тЖТ   188тЖТ  return `${days}d ago`;
   189тЖТ   189тЖТ}
