   121→function FadeIn({
   122→  children,
   123→  className,
   124→  delay = 0,
   125→}: {
   126→  children: React.ReactNode;
   127→  className?: string;
   128→  delay?: number;
   129→}) {
   130→  return (
   131→    <motion.div
   132→      initial={{ opacity: 0, y: 24 }}
   133→      whileInView={{ opacity: 1, y: 0 }}
   134→      viewport={{ once: true, margin: '-60px' }}
   135→      transition={{ duration: 0.5, delay, ease: 'easeOut' }}
   136→      className={className}
   137→    >
   138→      {children}
   139→    </motion.div>
   140→  );
   141→}
