    57→function downloadCSV(data: string[][], filename: string) {
    58→  const csv = data.map(row => row.map(cell => `"${cell}"`).join(',')).join('\n');
    59→  const blob = new Blob([csv], { type: 'text/csv' });
    60→  const url = URL.createObjectURL(blob);
    61→  const a = document.createElement('a');
    62→  a.href = url;
    63→  a.download = filename;
    64→  a.click();
    65→  URL.revokeObjectURL(url);
    66→}

   228→function timeAgo(iso: string) {
   229→  const diff = Date.now() - new Date(iso).getTime();
   230→  const mins = Math.floor(diff / 60000);
   231→  if (mins < 60) return `${mins}m ago`;
   232→  const hrs = Math.floor(mins / 60);
   233→  if (hrs < 24) return `${hrs}h ago`;
   234→  const days = Math.floor(hrs / 24);
   235→  return `${days}d ago`;
   236→}
